这是一个完整的系统性修改方案。我们将从“低维参数化矩阵计算”迁移到“高维像素级 FFT 计算”。



### 核心变更总结



1. **数学架构**：$c_{\mathbf{k}}$ 直接等于像素值。算子从矩阵乘法 $A = \Phi G$ 变为无矩阵运算 $A(x) = \Phi \cdot \text{FFT}(x)$。
2. **维度升级**：从 $21 \times 21$ (441) 升级为 $128 \times 128$ (16384)。
3. **计算策略**：使用 **Matrix-Free FFT** 替代 $O(N^2)$ 的稠密矩阵乘法，防止 4GB+ 的矩阵显存溢出；使用 **共轭梯度法 (CG)** 替代直接求逆来生成 Tikhonov 初值。

```
# image_generator.py
import torch
import torch.nn as nn
import torch.nn.functional as F

class DifferentiableImageGenerator(nn.Module):
    """
    进化版：1:1 像素映射生成器。
    当系数网格 (128x128) 等于图像尺寸时，系数即为图像本身。
    """
    def __init__(self, image_size: int = 128, coeff_grid=(128, 128)):
        super().__init__()
        self.H = image_size
        self.W = image_size
        # 允许传入 tuple 或 int
        if isinstance(coeff_grid, int):
            self.grid_H = self.grid_W = coeff_grid
        else:
            self.grid_H, self.grid_W = coeff_grid
        
        self.is_identity = (self.H == self.grid_H) and (self.W == self.grid_W)

    def forward(self, coeff_batch: torch.Tensor) -> torch.Tensor:
        # 输入形状可能是 (B, 1, 128, 128) 或 (B, 16384)
        if coeff_batch.dim() == 2:
            b = coeff_batch.shape[0]
            coeff_batch = coeff_batch.view(b, 1, self.grid_H, self.grid_W)
        
        # 如果是 1:1 映射，直接返回（零开销）
        if self.is_identity:
            return coeff_batch
            
        # 否则使用双线性插值（兜底）
        return F.interpolate(coeff_batch, size=(self.H, self.W), 
                             mode='bilinear', align_corners=True)
```

### 3. 修改 `radon_transform.py` (核心：Matrix-Free FFT & CG)



**目标**：实现 FFT 算子和 CG 求解器，彻底移除 $G$ 矩阵构建过程。

Python



```
# radon_transform.py
import torch
import numpy as np
from config import device, DATA_CONFIG, THEORETICAL_CONFIG, IMAGE_SIZE
from box_spline import CardinalBSpline2D
from image_generator import DifferentiableImageGenerator

def cg_solve(operator_func, adjoint_func, F_observed, x_init=None, n_iter=15, lambda_reg=0.1):
    """Matrix-Free 共轭梯度法求解 Tikhonov 问题"""
    # RHS = A^H * F
    rhs = adjoint_func(F_observed)
    
    if x_init is None:
        x = torch.zeros_like(rhs)
    else:
        x = x_init.clone()

    def A_tikhonov(v):
        # (A^H A + lambda I) v
        return adjoint_func(operator_func(v)) + lambda_reg * v

    r = rhs - A_tikhonov(x)
    p = r.clone()
    rsold = torch.sum(torch.abs(r) ** 2)

    for i in range(n_iter):
        Ap = A_tikhonov(p)
        # 避免除以零
        alpha_denom = torch.sum(torch.conj(p) * Ap).real
        if alpha_denom < 1e-12: break
            
        alpha = rsold / alpha_denom
        x = x + alpha * p
        r = r - alpha * Ap
        rsnew = torch.sum(torch.abs(r) ** 2)
        if torch.sqrt(rsnew) < 1e-6:
            break
        p = r + (rsnew / rsold) * p
        rsold = rsnew
    return x

class FourierOperatorCalculator:
    """计算对角阵 Phi 和排序索引，不再计算 G 矩阵"""
    def __init__(self, beta=(1, 128), n_coefficients=128*128, m=2):
        self.beta = torch.tensor(beta, dtype=torch.float32)
        self.n_coefficients = n_coefficients
        self._compute_k_order()
        self._compute_Phi_matrix()

    def _compute_Phi_matrix(self):
        N = self.n_coefficients
        j = torch.arange(N, dtype=torch.float64)
        # 频率坐标计算
        arg_x = (2.0 * np.pi * j / N) * self.beta[0].double()
        arg_y = (2.0 * np.pi * j / N) * self.beta[1].double()
        
        bspline = CardinalBSpline2D()
        B1_x = np.asarray(bspline.B1_hat_complex(arg_x.numpy()), dtype=np.complex128)
        B1_y = np.asarray(bspline.B1_hat_complex(arg_y.numpy()), dtype=np.complex128)
        
        # Phi = B^2 (因为是 Hat function 的卷积)
        Phi = B1_x * B1_y
        self.Phi_diagonal = torch.from_numpy(Phi).to(torch.complex128)

    def _compute_k_order(self):
        side = int(np.sqrt(self.n_coefficients))
        xs = torch.arange(side, dtype=torch.float64)
        ys = torch.arange(side, dtype=torch.float64)
        self.X, self.Y = torch.meshgrid(xs, ys, indexing="xy")
        beta_vec = self.beta.double()
        # 根据 beta 方向对频率进行排序
        beta_dot = (beta_vec[0] * self.X + beta_vec[1] * self.Y).reshape(-1)
        self.flatten_order = torch.argsort(beta_dot).to(torch.long)

class TheoreticalDataGenerator:
    """生成器：使用 FFT 进行前向物理模拟，使用 CG 或 Adjoint 进行初始化"""
    def __init__(self):
        self.img_size = IMAGE_SIZE
        self.N = self.img_size * self.img_size
        self.noise_level = DATA_CONFIG['noise_level']
        self.image_gen = DifferentiableImageGenerator(image_size=self.img_size).to(device)
        self.fourier_calc = FourierOperatorCalculator(
            beta=THEORETICAL_CONFIG['beta_vector'], 
            n_coefficients=self.N
        )
        self.Phi = self.fourier_calc.Phi_diagonal.to(device=device, dtype=torch.complex64)
        self.flatten_order = self.fourier_calc.flatten_order.to(device)

    def forward_operator(self, coeff_matrix):
        """A * d = Phi * FFT(sort(d))"""
        b = coeff_matrix.shape[0]
        c_flat = coeff_matrix.view(b, -1)
        d_ordered = c_flat[:, self.flatten_order].to(torch.complex64)
        # G 算子等价于 FFT
        Gd = torch.fft.fft(d_ordered, dim=-1)
        return Gd * self.Phi.unsqueeze(0)

    def adjoint_operator(self, residual):
        """A^H * r = unsort(IFFT(Phi^H * r) * N)"""
        # Phi^H * r
        PhiH_R = residual * self.Phi.conj().unsqueeze(0)
        # G^H 等价于 IFFT * N (因为 torch.fft 是非归一化的)
        d_complex = torch.fft.ifft(PhiH_R, dim=-1) * self.N
        d_real = d_complex.real
        
        # 逆排序 (scatter)
        b = residual.shape[0]
        out = torch.zeros(b, self.N, device=residual.device, dtype=d_real.dtype)
        out.scatter_(1, self.flatten_order.unsqueeze(0).expand(b, -1), d_real)
        return out.view(b, 1, self.img_size, self.img_size)

    def generate_training_sample(self, random_seed=None, lambda_reg=None):
        if random_seed is not None:
            torch.manual_seed(random_seed)
        
        # 1. 生成真实图像 (系数即像素)
        coeff_true = torch.rand(1, 1, self.img_size, self.img_size, device=device)
        # 简单的形状或随机噪声
        mask = torch.zeros_like(coeff_true)
        mask[:, :, 30:100, 30:100] = 1.0 # 简单的方块，便于观察
        coeff_true = coeff_true * mask

        # 2. 正向物理投影 F = A(d)
        F_clean = self.forward_operator(coeff_true)

        # 3. 加噪
        if self.noise_level > 0:
            noise = torch.randn_like(F_clean) + 1j * torch.randn_like(F_clean)
            scale = self.noise_level * torch.mean(torch.abs(F_clean))
            F_observed = F_clean + scale * noise
        else:
            F_observed = F_clean

        # 4. 初始化求解 (使用 CG)
        lam = lambda_reg if lambda_reg else DATA_CONFIG.get("lambda_reg", 0.1)
        # 如果追求速度，可以用 coeff_initial = self.adjoint_operator(F_observed) * 1e-4
        coeff_initial = cg_solve(self.forward_operator, self.adjoint_operator, 
                               F_observed, n_iter=10, lambda_reg=lam)
        
        return coeff_true.squeeze(0), coeff_true.squeeze(0), F_observed.squeeze(0), coeff_initial.squeeze(0)

    def generate_batch(self, batch_size, random_seed=None, lambda_reg=None):
        c_list, f_list, F_list, i_list = [], [], [], []
        start_seed = random_seed if random_seed is not None else 0
        for i in range(batch_size):
            c, f, F, init = self.generate_training_sample(start_seed + i, lambda_reg)
            c_list.append(c); f_list.append(f); F_list.append(F); i_list.append(init)
        return torch.stack(c_list), torch.stack(f_list), torch.stack(F_list), torch.stack(i_list)
```

```
import torch
import torch.nn as nn
from config import device, THEORETICAL_CONFIG
from radon_transform import FourierOperatorCalculator

class RadonFourierOperator2D(nn.Module):
    """基于 FFT 的无矩阵算子 A = Phi * FFT"""
    def __init__(self, beta=(1, 128), height=128, width=128, m=2):
        super().__init__()
        self.height = height
        self.width = width
        self.N = height * width
        
        calc = FourierOperatorCalculator(beta, self.N, m)
        self.register_buffer('Phi', calc.Phi_diagonal.to(device, dtype=torch.complex64))
        self.register_buffer('flatten_order', calc.flatten_order.to(device, dtype=torch.long))

    def forward(self, coeff_matrix):
        b = coeff_matrix.shape[0]
        flat = coeff_matrix.view(b, -1)
        ordered = flat[:, self.flatten_order].to(torch.complex64)
        return torch.fft.fft(ordered, dim=-1) * self.Phi.unsqueeze(0)

    def adjoint(self, residual):
        if residual.dim() == 3: residual = residual.squeeze(1)
        PhiH_R = residual * self.Phi.conj().unsqueeze(0)
        # IFFT * N 对应 G^H
        d_real = (torch.fft.ifft(PhiH_R, dim=-1) * self.N).real
        
        # Unsort / Scatter
        b = residual.shape[0]
        out = torch.zeros(b, self.N, device=residual.device, dtype=d_real.dtype)
        out.scatter_(1, self.flatten_order.unsqueeze(0).expand(b, -1), d_real)
        return out.view(b, 1, self.height, self.width)
```
