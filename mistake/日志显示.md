目的： 让验证过程除了返回总 Loss，也单独返回纯系数 Loss。

```
def _validate(self):
    """执行验证：每次随机生成验证批次"""
    # ... (前面的代码保持不变) ...
    
    coeff_true_val = coeff_true_val.to(device)
    F_observed_val = F_observed_val.to(device)
    coeff_initial_val = coeff_initial_val.to(device)
    
    # ... (中间代码保持不变) ...

    self.model.eval()
    with torch.no_grad():
        coeff_pred, history, metrics = self.model(coeff_initial_val, F_observed_val)
        
        # 计算系数损失
        loss_coeff = nn.functional.mse_loss(coeff_pred, coeff_true_val)
        
        image_loss_weight = TRAINING_CONFIG.get('image_loss_weight', 0.0)
        if image_loss_weight > 0:
            f_true_img = self.image_gen(coeff_true_val)
            f_pred_img = self.image_gen(coeff_pred)
            loss_image = nn.functional.mse_loss(f_pred_img, f_true_img)
            val_loss = loss_coeff + image_loss_weight * loss_image
        else:
            val_loss = loss_coeff
            
    self.model.train()
    
    # 【修改点】：增加返回 loss_coeff.item()
    return val_loss.item(), loss_coeff.item(), metrics 
```
